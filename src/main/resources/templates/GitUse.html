#创建分支
git branch 分支名
$ git branch -v
* master 8bd81ce 第三次提交
HUAWEI@Yv-Cheung MINGW64 /d/swDev/workSpaceForGit/testRpty (master)
$ git branch hot-fix#创建一个分支
HUAWEI@Yv-Cheung MINGW64 /d/swDev/workSpaceForGit/testRpty (master)
$ git branch -v#查看分支，此时有两个分支
  hot-fix 8bd81ce 第三次提交
* master  8bd81ce 第三次提交#当前分支
#查看分支
git branch -v 
$ git branch -v
* master 8bd81ce 第三次提交#*代表当前分支
#合并分支到当前分支
git merge 分支名
HUAWEI@Yv-Cheung MINGW64 /d/swDev/workSpaceForGit/testRpty (master)
$ git merge hot-fix

#切换分支
git checkout 分支名
#合并冲突分支，冲突的原因就是你创建一个分支后，在你合并分支前，主分支的一个内容已经修改了
#演示

#此时使用 git commit 命令时不能带文件名

远程仓库命令

xxxxxxxxxx
#查看当前的远程库
git remote -v
#添加远程库
git remote add 别名 远程地址  
#推送
git push 别名/链接 分支
#克隆
git clone 远程地址
